#!/usr/bin/env bash
set -euo pipefail

############################################
# Deploy Menu Script (fixed config in script)
############################################

# Default branch (edit if needed)
BRANCH="main"  # <-- đổi branch tại đây nếu cần

# ===== Fixed config inside script =====
APP_DIR="/home/laborant/xxl-job"
REPO_URL="https://github.com/Thisorp/xxl-job.git"
COMPOSE_PATH="/home/laborant/xxl-job/docker/docker-compose.yml"

LOG_TAIL_DEFAULT=200
HEALTH_TIMEOUT_DEFAULT=120

############################################
# Helpers
############################################
info(){ echo -e "\n[INFO] $*"; }
warn(){ echo -e "\n[WARN] $*" >&2; }
err(){  echo -e "\n[ERROR] $*" >&2; }
pause(){ read -rp $'\nNhấn Enter để tiếp tục...'; }

need_cmd() { command -v "$1" >/dev/null 2>&1 || { err "Thiếu command: $1"; return 1; }; }

mask_url() { sed -E 's#(https?://)[^/@]+@#\1***@#g'; }

detect_compose() {
  if docker compose version >/dev/null 2>&1; then
    echo "docker compose"
  elif command -v docker-compose >/dev/null 2>&1; then
    echo "docker-compose"
  else
    echo ""
  fi
}

ensure_compose_path() {
  if [ -f "${COMPOSE_PATH}" ]; then
    return 0
  fi
  err "Không tìm thấy compose file: ${COMPOSE_PATH}"
  err "Hãy sửa COMPOSE_PATH trong script cho đúng."
  return 1
}

ensure_prereqs() {
  need_cmd git || return 1
  need_cmd docker || return 1
  DC="$(detect_compose)"
  if [ -z "${DC}" ]; then
    err "Không tìm thấy docker compose (docker compose hoặc docker-compose)."
    return 1
  fi
  return 0
}

is_git_repo() {
  [ -d "${APP_DIR}/.git" ] && git -C "$APP_DIR" rev-parse --is-inside-work-tree >/dev/null 2>&1
}

get_remote_url() {
  git -C "$APP_DIR" remote get-url origin 2>/dev/null || true
}

need_git_creds_if_https() {
  local remote_url="$1"
  if [[ "$remote_url" =~ ^https?:// ]]; then
    read -rp "Git username: " GIT_USER
    read -rsp "Git password/token: " GIT_PASS
    echo ""
    export GIT_USER GIT_PASS
  else
    export GIT_USER="" GIT_PASS=""
  fi
}

auth_url_for_https() {
  local url="$1"
  local user="$2"
  local pass="$3"
  echo "$url" | sed -E "s#^(https?://)#\\1${user}:${pass}@#"
}

############################################
# Actions
############################################

action_clone_or_pull() {
  ensure_prereqs || { pause; return; }

  # Nếu đã là git repo -> pull
  if is_git_repo; then
    info "Repo đã tồn tại tại: $APP_DIR"
    local remote_url
    remote_url="$(get_remote_url)"
    if [ -z "$remote_url" ]; then
      err "Không lấy được remote origin URL."
      pause; return
    fi

    info "Remote: $(echo "$remote_url" | mask_url)"
    info "Pull branch: $BRANCH"

    need_git_creds_if_https "$remote_url"

    if [[ "$remote_url" =~ ^https?:// ]]; then
      local auth_url
      auth_url="$(auth_url_for_https "$remote_url" "$GIT_USER" "$GIT_PASS")"
      git -C "$APP_DIR" fetch "$auth_url" "$BRANCH"
      git -C "$APP_DIR" checkout "$BRANCH"
      git -C "$APP_DIR" reset --hard FETCH_HEAD
    else
      git -C "$APP_DIR" fetch origin "$BRANCH"
      git -C "$APP_DIR" checkout "$BRANCH"
      git -C "$APP_DIR" reset --hard "origin/${BRANCH}"
    fi

    info "Updated. Current commit: $(git -C "$APP_DIR" rev-parse --short HEAD)"
    pause
    return
  fi

  # Chưa có repo -> clone
  if [ -d "$APP_DIR" ] && [ "$(ls -A "$APP_DIR" 2>/dev/null | wc -l | tr -d ' ')" != "0" ]; then
    err "APP_DIR tồn tại nhưng không phải git repo và không rỗng: $APP_DIR"
    warn "Gợi ý: xóa/move thư mục này hoặc đổi APP_DIR trong script."
    pause
    return
  fi

  mkdir -p "$APP_DIR"
  info "Repo chưa tồn tại -> Cloning..."
  info "Repo URL: $(echo "$REPO_URL" | mask_url)"
  git clone "$REPO_URL" "$APP_DIR"

  info "Checkout branch: $BRANCH"
  git -C "$APP_DIR" checkout "$BRANCH" >/dev/null 2>&1 || true

  info "Clone xong. Current commit: $(git -C "$APP_DIR" rev-parse --short HEAD)"
  pause
}

action_compose_up() {
  ensure_prereqs || { pause; return; }
  ensure_compose_path || { pause; return; }

  info "Compose file: $COMPOSE_PATH"
  info "Chạy docker compose up -d (không pull images)..."
  $DC -f "$COMPOSE_PATH" up -d --remove-orphans

  info "DONE: compose up"
  pause
}

action_status() {
  ensure_prereqs || { pause; return; }
  ensure_compose_path || { pause; return; }

  info "=== docker compose ps ==="
  $DC -f "$COMPOSE_PATH" ps

  mapfile -t CIDS < <($DC -f "$COMPOSE_PATH" ps -q)
  if [ "${#CIDS[@]}" -eq 0 ]; then
    warn "Không thấy container nào thuộc compose stack."
    pause; return
  fi

  info "=== State/Health ==="
  printf "%-45s | %-10s | %-15s\n" "CONTAINER" "STATE" "HEALTH"
  printf -- "------------------------------------------------------------------------------------------\n"
  for cid in "${CIDS[@]}"; do
    name="$(docker inspect -f '{{.Name}}' "$cid" | sed 's#^/##')"
    state="$(docker inspect -f '{{.State.Status}}' "$cid")"
    health="$(docker inspect -f '{{if .State.Health}}{{.State.Health.Status}}{{else}}no-healthcheck{{end}}' "$cid")"
    printf "%-45s | %-10s | %-15s\n" "$name" "$state" "$health"
  done

  pause
}

action_wait_health() {
  ensure_prereqs || { pause; return; }
  ensure_compose_path || { pause; return; }

  read -rp "Health timeout seconds [${HEALTH_TIMEOUT_DEFAULT}]: " t
  local timeout="${t:-$HEALTH_TIMEOUT_DEFAULT}"

  mapfile -t CIDS < <($DC -f "$COMPOSE_PATH" ps -q)
  if [ "${#CIDS[@]}" -eq 0 ]; then
    warn "Không thấy container nào thuộc compose stack."
    pause; return
  fi

  info "Chờ container running/healthy (max ${timeout}s)..."
  local start_ts now_ts all_ok state health
  start_ts=$(date +%s)

  while true; do
    all_ok=true
    for cid in "${CIDS[@]}"; do
      state="$(docker inspect -f '{{.State.Status}}' "$cid")"
      health="$(docker inspect -f '{{if .State.Health}}{{.State.Health.Status}}{{else}}no-healthcheck{{end}}' "$cid")"
      [ "$state" = "running" ] || all_ok=false
      if [ "$health" != "no-healthcheck" ] && [ "$health" != "healthy" ]; then
        all_ok=false
      fi
    done

    $all_ok && { info "OK: Tất cả container running (và healthy nếu có healthcheck)."; break; }

    now_ts=$(date +%s)
    if (( now_ts - start_ts > timeout )); then
      warn "Timeout. Hãy xem Status/Logs để debug."
      break
    fi
    sleep 3
  done

  pause
}

action_logs() {
  ensure_prereqs || { pause; return; }
  ensure_compose_path || { pause; return; }

  read -rp "Tail lines [${LOG_TAIL_DEFAULT}]: " t
  local tail="${t:-$LOG_TAIL_DEFAULT}"

  echo ""
  echo "1) Logs tất cả services (tail)"
  echo "2) Logs follow (-f) tất cả services"
  echo "3) Logs 1 service (tail)"
  echo "4) Logs 1 service (follow)"
  echo "5) Logs grep theo keyword (all services, tail)"
  echo "6) Logs grep theo keyword (1 service, tail)"
  echo "7) Logs follow + grep keyword (all services)"
  echo "8) Logs follow + grep keyword (1 service)"
  read -rp "Chọn: " mode

  case "$mode" in
    1) $DC -f "$COMPOSE_PATH" logs --tail "$tail" ;;
    2) $DC -f "$COMPOSE_PATH" logs -f --tail "$tail" ;;
    3)
      $DC -f "$COMPOSE_PATH" config --services
      read -rp "Nhập service name: " svc
      [ -n "$svc" ] || { warn "Service name không được để trống."; pause; return; }
      $DC -f "$COMPOSE_PATH" logs --tail "$tail" "$svc"
      ;;
    4)
      $DC -f "$COMPOSE_PATH" config --services
      read -rp "Nhập service name: " svc
      [ -n "$svc" ] || { warn "Service name không được để trống."; pause; return; }
      $DC -f "$COMPOSE_PATH" logs -f --tail "$tail" "$svc"
      ;;
    5)
      read -rp "Nhập keyword cần grep: " keyword
      [ -n "$keyword" ] || { warn "Keyword không được để trống."; pause; return; }
      $DC -f "$COMPOSE_PATH" logs --tail "$tail" | grep -i --color=auto "$keyword" || true
      ;;
    6)
      $DC -f "$COMPOSE_PATH" config --services
      read -rp "Nhập service name: " svc
      read -rp "Nhập keyword cần grep: " keyword
      if [ -z "$svc" ] || [ -z "$keyword" ]; then
        warn "Service và keyword không được để trống."
        pause; return
      fi
      $DC -f "$COMPOSE_PATH" logs --tail "$tail" "$svc" | grep -i --color=auto "$keyword" || true
      ;;
    7)
      read -rp "Nhập keyword cần grep: " keyword
      [ -n "$keyword" ] || { warn "Keyword không được để trống."; pause; return; }
      $DC -f "$COMPOSE_PATH" logs -f --tail "$tail" | grep -i --line-buffered --color=auto "$keyword" || true
      ;;
    8)
      $DC -f "$COMPOSE_PATH" config --services
      read -rp "Nhập service name: " svc
      read -rp "Nhập keyword cần grep: " keyword
      if [ -z "$svc" ] || [ -z "$keyword" ]; then
        warn "Service và keyword không được để trống."
        pause; return
      fi
      $DC -f "$COMPOSE_PATH" logs -f --tail "$tail" "$svc" | grep -i --line-buffered --color=auto "$keyword" || true
      ;;
    *) warn "Lựa chọn không hợp lệ." ;;
  esac

  pause
}

action_full_deploy() {
  # Full deploy: pull code -> compose up -> wait -> status/logs (KHÔNG docker pull)
  ensure_prereqs || { pause; return; }
  ensure_compose_path || { pause; return; }

  if ! is_git_repo; then
    err "Repo chưa được clone. Hãy chọn Clone trước."
    pause; return
  fi

  local remote_url
  remote_url="$(get_remote_url)"
  [ -n "$remote_url" ] || { err "Không lấy được remote origin URL."; pause; return; }

  info "Step 1/4: Pull code ($BRANCH)"
  need_git_creds_if_https "$remote_url"
  if [[ "$remote_url" =~ ^https?:// ]]; then
    local auth_url
    auth_url="$(auth_url_for_https "$remote_url" "$GIT_USER" "$GIT_PASS")"
    git -C "$APP_DIR" fetch "$auth_url" "$BRANCH"
    git -C "$APP_DIR" checkout "$BRANCH"
    git -C "$APP_DIR" reset --hard FETCH_HEAD
  else
    git -C "$APP_DIR" fetch origin "$BRANCH"
    git -C "$APP_DIR" checkout "$BRANCH"
    git -C "$APP_DIR" reset --hard "origin/${BRANCH}"
  fi
  info "Commit: $(git -C "$APP_DIR" rev-parse --short HEAD)"

  info "Step 2/4: docker compose up -d (không pull images)"
  $DC -f "$COMPOSE_PATH" up -d --remove-orphans

  info "Step 3/4: wait health"
  local timeout="${HEALTH_TIMEOUT_DEFAULT}"
  mapfile -t CIDS < <($DC -f "$COMPOSE_PATH" ps -q)
  if [ "${#CIDS[@]}" -gt 0 ]; then
    local start_ts now_ts all_ok state health
    start_ts=$(date +%s)
    while true; do
      all_ok=true
      for cid in "${CIDS[@]}"; do
        state="$(docker inspect -f '{{.State.Status}}' "$cid")"
        health="$(docker inspect -f '{{if .State.Health}}{{.State.Health.Status}}{{else}}no-healthcheck{{end}}' "$cid")"
        [ "$state" = "running" ] || all_ok=false
        if [ "$health" != "no-healthcheck" ] && [ "$health" != "healthy" ]; then
          all_ok=false
        fi
      done
      $all_ok && break
      now_ts=$(date +%s)
      (( now_ts - start_ts > timeout )) && { warn "Timeout healthcheck."; break; }
      sleep 3
    done
  fi

  info "Step 4/4: status + logs tail"
  $DC -f "$COMPOSE_PATH" ps
  $DC -f "$COMPOSE_PATH" logs --tail "$LOG_TAIL_DEFAULT"

  pause
}

action_restart_service() {
  ensure_prereqs || { pause; return; }
  ensure_compose_path || { pause; return; }

  echo ""
  echo "1) Restart ALL services"
  echo "2) Restart 1 service"
  read -rp "Chọn: " mode

  case "$mode" in
    1)
      info "Restart toàn bộ services..."
      $DC -f "$COMPOSE_PATH" restart
      ;;
    2)
      info "Danh sách services:"
      $DC -f "$COMPOSE_PATH" config --services
      read -rp "Nhập service name: " svc
      if [ -z "$svc" ]; then
        warn "Service name không được để trống."
      else
        info "Restart service: $svc"
        $DC -f "$COMPOSE_PATH" restart "$svc"
      fi
      ;;
    *) warn "Lựa chọn không hợp lệ." ;;
  esac

  pause
}

action_test_service() {
  ensure_prereqs || { pause; return; }
  ensure_compose_path || { pause; return; }

  echo ""
  echo "1) Test containers state/health (compose)"
  echo "2) Test HTTP endpoint (curl)"
  echo "3) Test 1 service port open (nc)"
  read -rp "Chọn: " mode

  case "$mode" in
    1)
      info "=== docker compose ps ==="
      $DC -f "$COMPOSE_PATH" ps

      mapfile -t CIDS < <($DC -f "$COMPOSE_PATH" ps -q)
      if [ "${#CIDS[@]}" -eq 0 ]; then
        err "Không thấy container nào thuộc compose stack."
        pause; return
      fi

      info "=== Result: state/health ==="
      printf "%-45s | %-10s | %-15s\n" "CONTAINER" "STATE" "HEALTH"
      printf -- "------------------------------------------------------------------------------------------\n"

      local failed=0
      for cid in "${CIDS[@]}"; do
        local name state health
        name="$(docker inspect -f '{{.Name}}' "$cid" | sed 's#^/##')"
        state="$(docker inspect -f '{{.State.Status}}' "$cid")"
        health="$(docker inspect -f '{{if .State.Health}}{{.State.Health.Status}}{{else}}no-healthcheck{{end}}' "$cid")"
        printf "%-45s | %-10s | %-15s\n" "$name" "$state" "$health"

        if [ "$state" != "running" ]; then
          failed=1
        fi
        if [ "$health" != "no-healthcheck" ] && [ "$health" != "healthy" ]; then
          failed=1
        fi
      done

      if [ "$failed" -eq 0 ]; then
        info "TEST PASS: Tất cả container running (và healthy nếu có healthcheck)."
      else
        warn "TEST FAIL: Có container không running/healthy. Hãy xem logs."
      fi
      ;;
    2)
      need_cmd curl || { pause; return; }
      read -rp "Nhập URL healthcheck (vd: http://127.0.0.1:8080/health): " url
      [ -n "$url" ] || { warn "URL không được để trống."; pause; return; }

      info "curl -> $url"
      # -sS: silent nhưng vẫn show error; -m: timeout; -o /dev/null: bỏ body; -w: in status code
      code="$(curl -sS -m 10 -o /dev/null -w '%{http_code}' "$url" || true)"

      if [[ "$code" =~ ^2[0-9][0-9]$ ]]; then
        info "TEST PASS: HTTP $code"
      else
        warn "TEST FAIL: HTTP $code"
      fi
      ;;
    3)
      need_cmd nc || { pause; return; }
      read -rp "Host [127.0.0.1]: " host
      host="${host:-127.0.0.1}"
      read -rp "Port: " port
      [ -n "$port" ] || { warn "Port không được để trống."; pause; return; }

      info "Check TCP connect: $host:$port"
      if nc -z -w 3 "$host" "$port"; then
        info "TEST PASS: Port open"
      else
        warn "TEST FAIL: Port closed/unreachable"
      fi
      ;;
    *)
      warn "Lựa chọn không hợp lệ."
      ;;
  esac

  pause
}

############################################
# Menu
############################################
print_header() {
  clear || true
  echo "=============================================="
  echo " Deploy Menu (fixed config)"
  echo " Branch default : ${BRANCH}"
  echo "=============================================="
  echo " APP_DIR       : ${APP_DIR}"
  echo " REPO_URL      : ${REPO_URL}"
  echo " COMPOSE_PATH  : ${COMPOSE_PATH}"
  echo "----------------------------------------------"
}

main_loop() {
  while true; do
    print_header
    echo "1) Clone/Pull code (auto)"
    echo "2) Docker compose: up -d"
    echo "3) Check status (ps + health)"
    echo "4) Wait health (optional)"
    echo "5) View logs (tail/follow/grep)"
    echo "6) Full deploy (pull code -> compose up -> wait -> status/logs)"
    echo "7) Restart service (all hoặc 1 service)"
    echo "8) Test service (health/http/port)"
    echo "0) Exit"
    echo ""
    read -rp "Chọn: " choice

    case "$choice" in
      1) action_clone_or_pull ;;
      2) action_compose_up ;;
      3) action_status ;;
      4) action_wait_health ;;
      5) action_logs ;;
      6) action_full_deploy ;;
      7) action_restart_service ;;
      8) action_test_service ;;
      0) info "Bye!"; exit 0 ;;
      *) warn "Lựa chọn không hợp lệ."; pause ;;
    esac
  done
}

main_loop
